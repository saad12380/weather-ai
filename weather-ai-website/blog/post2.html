Real-time Anomaly Detection in Weather Data Streams

Michael Rodriguez
April 22, 2024
6 min read
1,450 views

In the era of IoT and big data, weather monitoring systems generate millions of data points every second. The real challenge isn't just collecting this data, but processing it in real-time to detect anomalies that could indicate critical weather events.

Our platform handles over 10 million data points per second from various sources including satellites, weather stations, drones, and IoT sensors. This article explores our approach to real-time anomaly detection and its practical applications.

Real-time Anomaly Detection Code

Here's our core anomaly detection algorithm implemented in Python:

# Real-time anomaly detection pipeline
from kafka import KafkaConsumer
import numpy as np
from scipy import stats
import redis
import json
from datetime import datetime

class RealTimeAnomalyDetector:
    def __init__(self):
        # Kafka consumer for data streams
        self.consumer = KafkaConsumer(
            'weather-data',
            bootstrap_servers=['kafka1:9092', 'kafka2:9092'],
            value_deserializer=lambda x: json.loads(x.decode('utf-8')),
            group_id='anomaly-detection-group',
            auto_offset_reset='latest'
        )
        
        # Redis for real-time state management
        self.redis_client = redis.Redis(host='localhost', port=6379, db=0)
        
        # Online statistical models for different metrics
        self.models = {
            'temperature': {'n': 0, 'mean': 0, 'M2': 0},
            'pressure': {'n': 0, 'mean': 0, 'M2': 0},
            'humidity': {'n': 0, 'mean': 0, 'M2': 0},
            'wind_speed': {'n': 0, 'mean': 0, 'M2': 0}
        }
        
        # Alert thresholds (standard deviations)
        self.thresholds = {
            'temperature': 3.0,
            'pressure': 2.5,
            'humidity': 3.0,
            'wind_speed': 4.0
        }
    
    def update_statistics(self, metric, value):
        """Update statistics using Welford's online algorithm"""
        model = self.models[metric]
        
        n = model['n'] + 1
        delta = value - model['mean']
        mean = model['mean'] + delta / n
        delta2 = value - mean
        M2 = model['M2'] + delta * delta2
        
        # Update model
        model['n'] = n
        model['mean'] = mean
        model['M2'] = M2
        
        return model
    
    def calculate_z_score(self, metric, value):
        """Calculate z-score for anomaly detection"""
        model = self.models[metric]
        
        if model['n'] < 2:
            return 0  # Not enough data
        
        variance = model['M2'] / model['n']
        if variance <= 0:
            return 0
        
        std = np.sqrt(variance)
        z_score = abs((value - model['mean']) / std)
        
        return z_score
    
    def detect_anomaly(self, data_point):
        """Detect anomalies in incoming data"""
        anomalies = []
        
        for metric in ['temperature', 'pressure', 'humidity', 'wind_speed']:
            if metric in data_point:
                value = float(data_point[metric])
                
                # Update statistics
                model = self.update_statistics(metric, value)
                
                # Calculate z-score
                z_score = self.calculate_z_score(metric, value)
                
                # Check if anomaly
                if z_score > self.thresholds[metric]:
                    anomaly = {
                        'metric': metric,
                        'value': value,
                        'z_score': z_score,
                        'mean': model['mean'],
                        'std': np.sqrt(model['M2'] / model['n']) if model['n'] > 1 else 0,
                        'timestamp': datetime.now().isoformat(),
                        'location': data_point.get('location', 'unknown')
                    }
                    anomalies.append(anomaly)
        
        return anomalies
    
    def trigger_alert(self, anomaly):
        """Trigger alert for detected anomaly"""
        alert_message = {
            'type': 'weather_anomaly',
            'severity': self.calculate_severity(anomaly['z_score']),
            'metric': anomaly['metric'],
            'value': anomaly['value'],
            'expected_range': f"{anomaly['mean'] - 2*anomaly['std']:.1f} - {anomaly['mean'] + 2*anomaly['std']:.1f}",
            'location': anomaly['location'],
            'timestamp': anomaly['timestamp'],
            'confidence': min(99.9, anomaly['z_score'] * 20)  # Convert z-score to percentage
        }
        
        # Store in Redis for real-time access
        alert_key = f"alert:{anomaly['timestamp']}"
        self.redis_client.setex(alert_key, 3600, json.dumps(alert_message))
        
        # Publish to alert topic
        self.redis_client.publish('weather-alerts', json.dumps(alert_message))
        
        return alert_message
    
    def calculate_severity(self, z_score):
        """Calculate severity level based on z-score"""
        if z_score > 5.0:
            return 'critical'
        elif z_score > 4.0:
            return 'high'
        elif z_score > 3.0:
            return 'medium'
        else:
            return 'low'
    
    def process_stream(self):
        """Main processing loop"""
        print("Starting anomaly detection service...")
        
        for message in self.consumer:
            try:
                data_point = message.value
                
                # Detect anomalies
                anomalies = self.detect_anomaly(data_point)
                
                # Trigger alerts for detected anomalies
                for anomaly in anomalies:
                    alert = self.trigger_alert(anomaly)
                    print(f"Alert triggered: {alert}")
                
                # Log processing stats every 1000 messages
                if message.offset % 1000 == 0:
                    print(f"Processed {message.offset} messages, detected {len(anomalies)} anomalies")
                    
            except Exception as e:
                print(f"Error processing message: {e}")
                continue

# Initialize and run the detector
if __name__ == "__main__":
    detector = RealTimeAnomalyDetector()
    detector.process_stream()

Key Implementation Details

1. Stream Processing Architecture

We use Apache Flink for stateful stream processing, allowing us to maintain statistical models in-memory while processing continuous data streams.

2. Online Learning

Traditional batch ML models are replaced with online learning algorithms that update continuously as new data arrives, without retraining from scratch.

3. Edge Computing

For critical applications, we deploy lightweight models directly on edge devices. This reduces latency and ensures detection even during network outages.

4. Multi-channel Alert System

When an anomaly is detected, our system immediately triggers alerts through SMS, email, mobile push notifications, and API webhooks with configurable escalation policies.
